// ===== INICIALIZAÇÃO ===== document.addEventListener('DOMContentLoaded', function() { // Iniciar a aplicação window.escaletaApp = new EscaletaHub(); // Atualizar estatísticas em tempo real const scriptEditor = document.getElementById('scriptEditor'); if (scriptEditor) { scriptEditor.addEventListener('input', function() { if (window.escaletaApp) { window.escaletaApp.updateDocumentStats(); } }); } // Formatação automática do roteiro document.querySelectorAll('[data-format]').forEach(btn => { btn.addEventListener('click', function() { const format = this.getAttribute('data-format'); window.escaletaApp.applyFormat(format); }); }); }); // ===== EXTENSÕES ADICIONAIS ===== EscaletaHub.prototype.applyFormat = function(format) { const editor = document.getElementById('scriptEditor'); const selection = window.getSelection(); if (format === 'scene') { const sceneElement = document.createElement('div'); sceneElement.className = 'scene-heading'; sceneElement.contentEditable = 'true'; sceneElement.textContent = 'EXT. NOVA CENA - DIA'; editor.appendChild(sceneElement); sceneElement.focus(); } else if (format === 'character') { const characterElement = document.createElement('div'); characterElement.className = 'character'; characterElement.contentEditable = 'true'; characterElement.textContent = 'NOVO PERSONAGEM'; editor.appendChild(characterElement); characterElement.focus(); } else if (format === 'dialogue') { const dialogueElement = document.createElement('div'); dialogueElement.className = 'dialogue'; dialogueElement.contentEditable = 'true'; dialogueElement.textContent = 'Novo diálogo...'; editor.appendChild(dialogueElement); dialogueElement.focus(); } }; // Sistema de personagens EscaletaHub.prototype.loadCharacters = async function() { if (!this.currentProject) return; const charactersSnapshot = await db.collection('characters') .where('projectId', '==', this.currentProject.id) .get(); const charactersList = document.getElementById('charactersList'); charactersList.innerHTML = ''; charactersSnapshot.forEach(doc => { const character = doc.data(); const characterCard = document.createElement('div'); characterCard.className = 'character-card'; characterCard.style.cssText = ` background: white; padding: 1rem; border-radius: 8px; box-shadow: 0 2px 5px rgba(0,0,0,0.1); `; characterCard.innerHTML = `

${character.name}

Idade: ${character.age || 'Não informada'}

Descrição: ${character.description || 'Sem descrição'}

`; charactersList.appendChild(characterCard); }); }; EscaletaHub.prototype.addCharacter = function() { const name = prompt('Nome do personagem:'); if (!name) return; const age = prompt('Idade do personagem:'); const description = prompt('Descrição do personagem:'); db.collection('characters').add({ name: name, age: age, description: description, projectId: this.currentProject.id, createdAt: firebase.firestore.FieldValue.serverTimestamp() }).then(() => { this.loadCharacters(); this.showNotification('Personagem adicionado!', 'success'); }); }; // Compartilhamento de projetos EscaletaHub.prototype.shareProject = async function() { if (!this.currentProject) return; const email = prompt('Digite o e-mail da pessoa para compartilhar:'); if (!email) return; // Aqui você pode implementar o compartilhamento real // Por enquanto, vamos apenas simular this.showNotification(`Convite enviado para ${email}`, 'success'); }; // Sistema de notificações melhorado EscaletaHub.prototype.showNotification = function(message, type = 'info') { // Criar elemento de notificação const notification = document.createElement('div'); notification.style.cssText = ` position: fixed; top: 20px; right: 20px; padding: 15px 20px; background: ${type === 'success' ? '#41b883' : type === 'error' ? '#e74c3c' : '#2a5a78'}; color: white; border-radius: 8px; z-index: 10000; animation: slideIn 0.3s ease; max-width: 300px; box-shadow: 0 4px 15px rgba(0,0,0,0.2); `; notification.textContent = message; document.body.appendChild(notification); // Remover após 4 segundos setTimeout(() => { notification.style.animation = 'slideOut 0.3s ease'; setTimeout(() => { if (notification.parentNode) { notification.parentNode.removeChild(notification); } }, 300); }, 4000); }; // Adicionar estilos de animação para notificações const style = document.createElement('style'); style.textContent = ` @keyframes slideIn { from { transform: translateX(100%); opacity: 0; } to { transform: translateX(0); opacity: 1; } } @keyframes slideOut { from { transform: translateX(0); opacity: 1; } to { transform: translateX(100%); opacity: 0; } } `; document.head.appendChild(style);
